import { useState } from 'react';
import { Card, Form, Input, Button, Select, App } from 'antd';
import { PlusOutlined } from '@ant-design/icons';
import apiClient from '../services/api';
 
const { TextArea } = Input;
const { Option } = Select;
 
const Episodes = () => {
  const { message } = App.useApp();
  const [form] = Form.useForm();
  const [loading, setLoading] = useState(false);
 
  const handleSubmit = async (values: any) => {
    try {
      setLoading(true);
      const response = await apiClient.createEpisode(values);
      message.success(`Episode created: ${response.id}`);
      form.resetFields();
    } catch (error: any) {
      message.error(error.response?.data?.detail || 'Failed to create episode');
    } finally {
      setLoading(false);
    }
  };
 
  return (
    <div>
      <h1>Episodes</h1>
      <p>Create and manage episodes to build your knowledge graph</p>
      
      <Card title="Create New Episode" style={{ marginTop: 24 }}>
        <Form
          form={form}
          layout="vertical"
          onFinish={handleSubmit}
          initialValues={{
            source_type: 'text',
          }}
        >
          <Form.Item
            label="Episode Name"
            name="name"
            rules={[{ required: true, message: 'Please enter episode name' }]}
          >
            <Input placeholder="e.g., User Preference" />
          </Form.Item>
 
          <Form.Item
            label="Content"
            name="content"
            rules={[{ required: true, message: 'Please enter content' }]}
          >
            <TextArea
              rows={6}
              placeholder="Enter episode content (text or JSON)"
            />
          </Form.Item>
 
          <Form.Item
            label="Source Type"
            name="source_type"
          >
            <Select>
              <Option value="text">Text</Option>
              <Option value="json">JSON</Option>
            </Select>
          </Form.Item>
 
          <Form.Item
            label="Source Description (Optional)"
            name="source_description"
          >
            <Input placeholder="e.g., User conversation, API data" />
          </Form.Item>
 
          <Form.Item
            label="Group ID (Optional)"
            name="group_id"
          >
            <Input placeholder="e.g., user_123" />
          </Form.Item>
 
          <Form.Item>
            <Button
              type="primary"
              htmlType="submit"
              icon={<PlusOutlined />}
              loading={loading}
            >
              Create Episode
            </Button>
          </Form.Item>
        </Form>
      </Card>
    </div>
  );
};
 
export default Episodes;
